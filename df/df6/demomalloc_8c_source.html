60 OPHeap* GetOPHeap()
61 {
62  while (true)
63  {
64  if (heap)
65  return heap;
66  if (atomic_flag_test_and_set_explicit(&lock, memory_order_acquire))
67  continue;
68  OPHeapNew(&heap);
69  atomic_flag_clear_explicit(&lock, memory_order_release);
70  }
71 }
