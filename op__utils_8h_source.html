63 static inline bool memeq(const void* ptr1, const void* ptr2, size_t num)
64 {
65  uintptr_t p1, p2;
66  p1 = (uintptr_t)ptr1;

68 
69 #ifdef __SSE4_1__
70  for (; num >= 16; num -= 16, p1 += 16, p2 += 16)
71  {
72  __m128i d1 = _mm_lddqu_si128((void*)p1);
73  __m128i d2 = _mm_lddqu_si128((void*)p2);
74  __m128i cmp = _mm_cmpeq_epi64(d1, d2);
75  int result = _mm_movemask_pd((__m128d)cmp);
76  if (result != 0x3)
77  return false;

88  }

203 static inline int fftstr1 (uint32_t x, int n)
204 {
205  int s;
206  while (n > 1)
207  {
208  s = n >> 1;
209  x = x & (x >> s);
210  n = n - s;
211  }
212  if (x == 0) return -1;
213  return __builtin_ctz(x);
214 }
