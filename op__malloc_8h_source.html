429 static inline void
430 OPLenRefCreate(oplenref_t* ref, void* data, size_t data_size,
431  size_t container_size)
432 {
433  uintptr_t uintref = (uintptr_t)ref;
434  if (data_size > OPLENREF_MAX_LEN)
435  {
436  *ref = 0;
437  return;
438  }
439  if (data_size > container_size)
440  {
441  void* ptr;
442  ptr = OPCalloc(ObtainOPHeap(ref), 1, data_size);
443  if (!ptr)
444  {
445  *ref = 0;
446  return;
447  }
448  memcpy(ptr, data, data_size);
449  *ref = (oplenref_t)OPPtr2Ref(ptr);
450  }
451  else
452  {
453  memcpy((void*)(uintref + sizeof(oplenref_t)), data, data_size);
454  *ref = 0;
455  }
456  *ref |= data_size << OPHEAP_BITS;
457 }
